<program>       ::= <statement> \n [<program>]

<statement>     ::=
  "LET" <identifier> "=" <expr>
| "PRINT" <obj>
| "IF" <cond> "THEN" <command> "ELSE" <command>
| "GOTO" <identifier - label>
| "LABEL" <identifier - label>
| "GOSUB" <identifier - label>
| "RETURN"
| "PUSH" <obj>
| "PULL" <identifier>
| "END"
| "PEEK"  <identifier - address> <identifier - destination>
| "POKE"  <identifier - address> <identifier - source>
| "REM" <identifier - comment>*  
| "READ"

<cond>          ::= 
  <obj> == <obj>
| <obj> >  <obj>
| <obj> <  <obj>
| <obj> <> <obj>

<expr>          ::= <obj> <op> <obj>
<op>            ::= "+" | "-" | "*" | "/"
<obj>           :: <identifier> | <const>

<const>         ::= <digit>*
<identifier>    ::= <letter>*

<digit>         ::= "1" .. "9"
<letter>        ::= "a" .. "z" | "A" .. "Z"
